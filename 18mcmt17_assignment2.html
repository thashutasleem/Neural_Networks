def PCA(data,num_dims):
    print(data.shape)
    scaler = StandardScaler() 
    std_data = scaler.fit_transform(data)                      #standardizing
    mean_vec = np.mean(std_data, axis=0)
    cov_mat = np.cov(std_data.T)                                     #finding covariance matrix
    eig_vals, eig_vecs = np.linalg.eig(cov_mat)             #CALCULATING EIGEN VALUES AND VECTORS.  
    eig_pairs=EIG_PAIRS(eig_vals, eig_vecs)
    eig_vecs = np.array([eig_pairs[i][1] for i in range(num_dims)])
    
    pca_data=(std_data).dot(eig_vecs.T)
    print(pca_data.shape)
    return pca_data

FLDA ALGORITM:

1.calculating the with-in class covariance matrix and between class covariance matrix for all the classes /# S_B = { (class data - mean) (class data - mean).T} S_W = { (class mean - total mean) (class mean - total mean).T}

2.summing up the with-in class covariance matrix and between class covariance matrix /# S_B = S_B1 + S_B2 + S_B3... S_W = S_W1 + S_W2 + S_W3.. #/

def fld_params(data,total_mean):
    mean=np.mean(data,axis=0)
    
    # calculate within class covariance matrix S_B = \sigma{N_i (x - m) (x - m).T}                                                             
    s_w=((data-mean).T).dot(data-mean)    
    
    #   calculate between class covariance matrix S_W = \sigma{N_i (m_i - m) (m_i - m).T}
    s_b=((total_mean-mean).T).dot(total_mean-mean)
    return s_w,s_b


def FLDA(data):
    total_mean=np.mean(data,axis=0)
    s_w1,s_b1=fld_params(data[:50],total_mean)
    s_w2,s_b2=fld_params(data[50:100],total_mean)
    s_w3,s_b3=fld_params(data[100:150],total_mean)
    S_W=(s_w1+s_w2+s_w3)
    S_B=50*(s_b1+s_b2+s_b3)
    # objective : find eigenvalue, eigenvector pairs for inv(S_W).S_B
    mat = np.dot(np.linalg.pinv(S_W), S_B)
    eigvals, eigvecs = np.linalg.eig(mat)
    eig_pairs = [(eigvals[i], eigvecs[:, i]) for i in range(len(eigvals))]
    eig_pairs = sorted(eig_pairs, key = lambda x : x[0], reverse = True)
    fld_data=data.dot(eig_pairs[0][1])
    return fld_data

def scree_plot(data):
    no_of_features = data.shape[1]
    cov_mat = np.cov(data.T) 
    eig_vals, eigen_vecs = np.linalg.eig(cov_mat)
    eig_pairs=EIG_PAIRS(eig_vals, eig_vecs)
    eig_vals, eig_vecs = [x[0] for x in eig_pairs], [y[1] for y in eig_pairs]
    plt.plot(np.arange(len(eig_vals)) + 1, eig_vals, 'ro-')
    plt.title('Plotting  arcene eigens')
    plt.xlabel('Principal Component')
    plt.ylabel('Eigen Values')
    plt.show()
    return eig_vals, eig_vecs
